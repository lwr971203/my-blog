    • static_cast
      #include <iostream>
using namespace std;
class A {
public:
    int a = 100;

    void AShow() {
        cout << this->a << endl;
    }
};

class B : public A {
public:
    int a = 200;

    void BShow() {
        cout << this->a << endl;
    }
};

// static_cast 要求原类型和目标类型可以转换。如果转换失败将发生编译错误
void Test1() {
    int i1 = 0;
    double d1 = static_cast<int>(i1); // c 语言的隐式类型转换都可以使用 static_cast 来替换，并且编译期不会给出警告。因为目的很明确，精度损失对你来说是可接受的
    int* p1 = nullptr;
    // int i2 = static_cast<int>(p1); // 编译 error
    double d2 = 1.99;
    void* vptr = &d2; // void* 无类型指针
    double* p2 = static_cast<double*>(vptr);

    // 具有继承关系的安全转换都是必须存在子类实例的
    B* b = new B;
    static_cast<A*>(b)->AShow(); // 100 向上转型，天然安全（因为子类是在继承父类的空间上面开辟）
    A* a = new A;
    static_cast<B*>(a)->BShow(); // 垃圾值。向下转型，如果指向或引用的实例是父类，那么不安全
    A* dynamic = new B;
    static_cast<B*>(dynamic)->BShow(); // 200。向下转型，但指向或引用的实例是子类，那么是安全的
}


int main() {
    Test2();
}
    • const_cast
      #include <iostream>
using namespace std;

// const_cast：去掉源数据的 const 限定。如果想改源数据，仅当源数据有 volatile 限定的时候才能改
void Test1() {
    const int i1 = 10; // i1 放在寄存器中只读
    int* p1 = const_cast<int*>(&i1); // 去掉 const 。p1 指向的是内存中的副本值
    cout << *p1 << endl; // 10
    *p1 = 20;
    cout << *p1 << endl; // 20
    cout << i1 << endl; // 10 i1 并没有改变. 因为 i1 对象 与 p1 指向的不是同一个对象

    const volatile int i2 = 30; // i1 放在寄存器中可读可写
    int* p2 = const_cast<int*>(&i2); // 去掉 const p2 指向的是寄存器中的 i2
    cout << *p2 << endl; // 30
    *p2 = 40;
    cout << *p2 << endl; // 40
    cout << i2 << endl; // 40 i2 随之改变

    const int i3 = 50; // i3 放在寄存器中只读
    // int i4 = const_cast<int>(i3); // error 目标类型必须是指针或引用
    int i5 = const_cast<int&>(i3); // 去掉 const, i5 是内存中的一个副本值
    cout << i5 << endl;
    int& r1 = const_cast<int&>(i3); // 虽然 r1 是引用语义，但引用的是内存中的副本值
    cout << r1 << endl; // 50
    r1 = 60;
    cout << i5 << endl; // 60
    cout << i3  << endl; // 50 i3 并没有发生改变

    const volatile int i6 = 70; // i6 放在寄存器中可读可写
    int& r2 = const_cast<int&>(i6); // 去掉 const r2 引用的是寄存器中的 i6
    cout << r2 << endl; // 70
    r2 = 80;
    cout << r2 << endl; // 80
    cout << i6 << endl; // 80 i3 随之发生改变
}

int main() {
    Test1();
}
    • dynamic_cast
      #include <iostream>
using namespace std;

class A {
private:
    virtual void ShowInfo() {
        cout << "I'm father ..." << endl;
    }
};

class B : public A {
public:
    void ShowInfo() override { // 冷知识：对于私有的虚方法,派生类能用更高级的访问级别去实现和暴露
        cout << "I'm son ..." << endl;
    }
};

// dynamic_cast 发生在运行时，要求形成多态关系，并且转换关系要求是安全的。转换目标类型只能是指针或引用类型，如果失败了分别返回 nullptr 或抛 bad_cast 异常
void Test1() {
    A* dynamic = new B;
    B* b = dynamic_cast<B*>(dynamic); // 注：dynamic_cast 的目标类型必须是指针或引用类型
    if (b != nullptr) {
        b->ShowInfo();
    } else {
        cout << "转换失败!" << endl;
    }
}

int main() {
    Test1();
}
    • reinterpret_cast
      #include <iostream>
using namespace std;

class A {
public:
    int a;

    void Func1() {
        cout << "A Func1 ..." << endl;
    }
};

class B : public A {
public:
    int b;

    void Func2() {
        cout << "B Func2 ..." << endl;
    }
};

// reinterpret_cast 的转换规则比较玄学，而且对后续可能出现bug
void Test1() {
    int i1 = 30;
    int* pi1 = &i1;
    char* c1 = "hello"; // warning...
    A a;
    B b;

    int* pi3 = reinterpret_cast<int*>(c1); // ok
    int* pi2 = reinterpret_cast<int*>(i1); // ok
    char* pc1 = reinterpret_cast<char*>(&b); // ok
    A* pa1 = reinterpret_cast<A*>(&b); // ok 跟继承不继承没有关系
    A* pa2 = reinterpret_cast<A*>(pi1); // ok
    pa1->Func1();
    pa2->Func1();
    // pa1->Func2(); // error

    // long l1 = reinterpret_cast<long>(pa1); // error 丢精度，指针 8 位， long 为 4 位
    // int i2 = reinterpret_cast<int>(pi1); // error 丢精度，指针 8 位，int 为 4 位
    long long l1 = reinterpret_cast<long long>(pa2); // ok 这个很常用来得到原十六进制地址的十进制表达方式
    // float f1 = reinterpret_cast<float>(i1); // 编译 error 不允许将整数转成浮点数
    // B b1 = reinterpret_cast<B>(a); // error
    // A a1 = reinterpret_cast<A>(b); // error

    // 综上所述：向指针类型转可以转，指针可以向不低于指针内存大小的整数转，向值类型转一般不成功
}

void Test2() {
    B b;
    A* pa1 = &b;
    B* pb1 = reinterpret_cast<B*>(pa1);
    B* pb2 = static_cast<B*>(pa1);

    pb1->Func2();
    pb1->Func1();
    cout << pb1->a << endl; // 30
    cout << pb2->a << endl; // 30
}

void align(char* p) {
    *p = 'x';
    cout << "call align ..." << endl;
}
void Test3() {
    const char* str = "h";
    // align(str); // eror

    typedef void (*PF)(const char*);
    PF pf;
    // pf = align; // error 类型错误
    // pf = static_cast<PF>(align); // error 类型不匹配不允许转换
    pf = reinterpret_cast<PF>(align); // ok 可以看出 reinterpret_cast 的转换是相对不安全的
    pf(str); // 编译没有 error , 但运行时抛异常了。这其实是利用了 reinterpret_cast 去欺骗类型系统，妄想修改 const char*. 但这逃不过 c++ 的法眼
}

    Test1();
